on_10_years_rashidun = {
	effect = {
		if = {
			limit = {
				AND = {
					dummy_male.current_year < 800
					title:e_rashidun = {
						is_title_created = yes
					}
				}
			}
			title:e_rashidun.holder = {
				trigger_event = {
					id = islam.0005
					days = {20 30}
				}
			}
		}
		if = {
			limit = {
				AND = {
					dummy_male.current_year < 800
					title:e_rashidun = {
						is_title_created = yes
						any_title_to_title_neighboring_kingdom = {
							is_title_created = yes
							holder = {
								NOT = {
									has_religion = religion:islam_religion
								}
							}
						}
					}
					title:e_rashidun.holder = {
						is_ai = yes
						is_at_war = no
					}
				}
			}
			title:e_rashidun = {
				random_title_to_title_neighboring_kingdom = {
					limit = {
						is_title_created = yes
						holder = {
							NOT = {
								has_religion = religion:islam_religion
							}
							trigger_if = {
								limit = {
									is_independent_ruler = no
								}
								liege = {
									NOT = {
										has_religion = religion:islam_religion
									}
								}
							}
						}
					}
					title:e_rashidun.holder = {
						start_war = {
							cb = holy_jihad_cb 
							target = PREV.holder
							target_title = PREV
						}
					}
				}
			}
		}
	}
	on_actions = {
		delay = { years = 5 }
		on_10_years_rashidun
	}
}
